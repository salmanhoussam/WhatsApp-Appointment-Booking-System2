```js
// TopItemsWidget — تحسب من orders المحلية بدون endpoint جديد
function buildTopItems(orders) {
  const counts = {}
  orders.forEach(order => {
    (order.items ?? []).forEach(item => {
      const key = item.catalog_item_id ?? item.id
      if (!counts[key]) counts[key] = { ...item, qty: 0 }
      counts[key].qty += item.quantity ?? 1
    })
  })
  return Object.values(counts).sort((a, b) => b.qty - a.qty).slice(0, 5)
}
```

```js
// ActivityFeed — يدمج orders + reservations ويفرزهم بالتاريخ
function buildActivityFeed(orders, reservations) {
  const events = [
    ...orders.map(o => ({ type: 'order', data: o, at: o.created_at ?? o.createdAt })),
    ...reservations.map(r => ({ type: 'reservation', data: r, at: r.created_at ?? r.createdAt })),
  ]
  return events.sort((a, b) => new Date(b.at) - new Date(a.at)).slice(0, 10)
}
```
